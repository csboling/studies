\begin{frame}{Compressive Sensing}
  \begin{itemize}
    \item{Wish to find $\vec{\alpha}_{N}$ such that 
          $\vec{x}_{N} = \mathbf{\Psi} \vec{\alpha}$
          given 
          $$
          \vec{y}_{K} = \mathbf{\Phi}_{K \times N}\vec{x}_N 
                      = \mathbf{\Phi}_{K \times N}
                        \mathbf{\Psi}_{N \times N}
                        \vec{\alpha}_N
          $$
          where $K \ll N$.}
     \item{Possible if $\|\vec{\alpha}\|_0 \ll N$.}
     \item{Importantly, possible for any sparsifying $\Psi$ if $\mathbf{\Phi}$ is random.}
  \end{itemize}
\end{frame}
